  static private function addUserFromLdap($username)
  {
    
    try
    {
      $user=new sfGuardUser();
      $user->
      setUsername($username)->
      setSalt('foo')->
      save();
      
      $profile= new sfGuardUserProfile();
      $profile->
      setUserId($user->getId())->
      save();
      error_log(sprintf(
        'Wviola: Added user %s to DB',
        $username));
    }
    catch (Exception $e)
    {
      error_log(sprintf(
        'Wviola: Could not add user %s to DB',
        $username));
    }
    
  }
